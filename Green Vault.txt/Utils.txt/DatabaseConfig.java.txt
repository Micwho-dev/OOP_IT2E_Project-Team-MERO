package utils;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

/**
 * Database configuration utility for H2 database.
 * Provides connection strings and database settings.
 */
public class DatabaseConfig {
    // Database folder location
    private static final String DB_FOLDER = "./database";
    // Database file location (file-based, persistent)
    private static final String DB_PATH = DB_FOLDER + "/greenvault";
    
    // Connection settings
    public static final String JDBC_URL = "jdbc:h2:file:" + DB_PATH + ";AUTO_SERVER=TRUE;AUTO_SERVER_PORT=9092";
    public static final String DB_USER = "user";
    public static final String DB_PASSWORD = "password";
    public static final String DB_DRIVER = "org.h2.Driver";
    
    // Console settings
    public static final int CONSOLE_PORT = 8082;
    
    /**
     * Ensures the database folder exists.
     * Creates it if it doesn't exist.
     */
    public static void ensureDatabaseFolder() {
        java.io.File dbFolder = new java.io.File(DB_FOLDER);
        if (!dbFolder.exists()) {
            dbFolder.mkdirs();
            System.out.println("Created database folder: " + dbFolder.getAbsolutePath());
        }
    }
    
    /**
     * Gets the database folder path.
     * @return Database folder path
     */
    public static String getDatabaseFolder() {
        return DB_FOLDER;
    }
    
    /**
     * Gets a database connection (embedded mode, no server).
     * Use this for initialization and direct access.
     * @return Connection object
     * @throws SQLException if connection fails
     */
    public static Connection getConnection() throws SQLException {
        // Ensure database folder exists
        ensureDatabaseFolder();
        
        try {
            Class.forName(DB_DRIVER);
        } catch (ClassNotFoundException e) {
            throw new SQLException("H2 Driver not found. Make sure h2-2.2.224.jar is in classpath.", e);
        }
        // Use AUTO_SERVER mode to allow multiple connections (app + console server)
        // This prevents "file is locked" errors when console server is running
        // Add DB_CLOSE_DELAY=-1 to keep database open after last connection closes
        // Add IFEXISTS=FALSE to allow database creation if it doesn't exist
        String embeddedUrl = "jdbc:h2:file:" + DB_PATH + ";AUTO_SERVER=TRUE;AUTO_SERVER_PORT=9092;DB_CLOSE_DELAY=-1;IFEXISTS=FALSE";
        Connection conn = DriverManager.getConnection(embeddedUrl, DB_USER, DB_PASSWORD);
        // Set auto-commit to true for better compatibility
        conn.setAutoCommit(true);
        return conn;
    }
    
    /**
     * Gets a database connection with server mode enabled.
     * Use this when H2 console server is running.
     * @return Connection object
     * @throws SQLException if connection fails
     */
    public static Connection getConnectionWithServer() throws SQLException {
        // Ensure database folder exists
        ensureDatabaseFolder();
        
        try {
            Class.forName(DB_DRIVER);
        } catch (ClassNotFoundException e) {
            throw new SQLException("H2 Driver not found. Make sure h2-2.2.224.jar is in classpath.", e);
        }
        // Add DB_CLOSE_DELAY=-1 to keep database open
        String serverUrl = JDBC_URL + ";DB_CLOSE_DELAY=-1";
        Connection conn = DriverManager.getConnection(serverUrl, DB_USER, DB_PASSWORD);
        conn.setAutoCommit(true);
        return conn;
    }
    
    /**
     * Gets the JDBC connection URL for embedded mode.
     * @return JDBC URL string
     */
    public static String getJdbcUrl() {
        return JDBC_URL;
    }
    
    /**
     * Gets the JDBC connection URL for console mode.
     * @return JDBC URL string for console connection
     */
    public static String getConsoleJdbcUrl() {
        // For console connection via TCP, use relative path from baseDir
        // The TCP server's baseDir is set to the project root
        // So we use the relative path: ./data/greenvault
        // Replace backslashes with forward slashes for JDBC URL
        String dbPath = DB_PATH.replace("\\", "/");
        
        // For console connection via TCP, allow automatic database creation
        // IFEXISTS=FALSE allows creation if database doesn't exist
        // DB_CLOSE_DELAY=-1 keeps database open
        // Use relative path since baseDir is set in TCP server
        return "jdbc:h2:tcp://localhost:9092/" + dbPath + ";IFEXISTS=FALSE;DB_CLOSE_DELAY=-1";
    }
    
    /**
     * Gets the JDBC connection URL for console mode using absolute path.
     * Use this if relative path doesn't work.
     * @return JDBC URL string for console connection with absolute path
     */
    public static String getConsoleJdbcUrlAbsolute() {
        // Get the absolute path to the database file
        java.io.File dbFile = new java.io.File(DB_PATH);
        String absolutePath = dbFile.getAbsolutePath();
        // Replace backslashes with forward slashes for JDBC URL
        absolutePath = absolutePath.replace("\\", "/");
        
        // For console connection via TCP, allow automatic database creation
        return "jdbc:h2:tcp://localhost:9092/" + absolutePath + ";IFEXISTS=FALSE;DB_CLOSE_DELAY=-1";
    }
    
    /**
     * Gets an alternative JDBC URL for console (embedded mode).
     * Use this if TCP connection doesn't work.
     * @return JDBC URL string for direct file connection
     */
    public static String getEmbeddedJdbcUrl() {
        return JDBC_URL.replace(";AUTO_SERVER=TRUE;AUTO_SERVER_PORT=9092", "");
    }
    
    /**
     * Gets the database username.
     * @return username
     */
    public static String getUsername() {
        return DB_USER;
    }
    
    /**
     * Gets the database password.
     * @return password
     */
    public static String getPassword() {
        return DB_PASSWORD;
    }
}
